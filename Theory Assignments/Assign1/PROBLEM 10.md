#PROBLEM 10 PSEUDOCODE

  input num
  
  digit <-- 0
  
  value <-- ""
  
  while num > 0 Do
  
      digit = num / 10
      
  if digit == 9 then
  
      value = value + string(digit)
      num = num / 10
      
  else 
  
      value = value + string(digit)
      num = num / 10
      
  end if 
  
  end while
  
  output value
  
  end
